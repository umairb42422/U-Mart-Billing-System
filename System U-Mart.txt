#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main(){
    int item;
    float pri;
    float tdis;
    float dis;
    float  pribdis;
    float priadis;
    int cash;
    float change;
    float gst=5;
    int tgst;
    cout<<"------- Welcome to U-MART------"<<endl;
    cout<<endl;
    cout<<"       ---[ RECEIPT ]---           "<<endl;
    cout<<endl;
    cout<<"(For Dealer)"<<endl;
    cout<<"Enter the no.of items        :  ";
    cin>>item;
    cout<<"Enter the price              :  ";
    cin>>pri;
    cout<<"Enter the discount %         :  ";
    cin>>dis;
    cout<<"------------------------------"<<endl;
    
    cout<<"(For Customer)"<<endl;
